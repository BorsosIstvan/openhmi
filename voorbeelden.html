if (newType === "button") { currentProject.objects[index] = { ...base, label: "Knop", publishTopic: "", publishPayload: "" }; } else if (newType === "led") { currentProject.objects[index] = { ...base, subscribeTopic: "", stateOn: "aan", stateOff: "uit" };
